C

Below is a sample C code from a magnet switch used on the Elevator Project. To see the Elevator Project go to the Project Tab on the left and go to Featured.



//ENGR 102 -03 Brian Gray
//Program: magnet_switch.c

#include
#include

void MAGNET(char* magnet){
//void function using a character so there can be multiple forms of this one function

pinMode(1, OUTPUT);
pinMode(2, OUTPUT);
//sets pins 1 and 2 as sending out current for an RGB LED

if(magnet == "OFF"){
digitalWrite(1, HIGH);
digitalWrite(2, LOW);
//If the magnet is off one color of the LED will turn off and the other will turn on
}
if(magnet == "ON") {
digitalWrite(1, LOW);
digitalWrite(2, HIGH);
//If the magnet is off then the other color of the LED will turn on
}
}

int main () {

wiringPiSetup();
//necessary for using the 40 pin connector
pinMode(6, INPUT);
//sets pin 6 to read in current (either on or off)

MAGNET("OFF");
//starts the program with one light being off signalling the magnet sensor is off

while(1) {
//Type of infinite loop

if(1 == digitalRead(6)) {
delay(20);
//if the magnet sensor is on then wait 20 miliseconds. This makes sure that there isn't a
//burst of magnetic flux somewhere
if(1 == digitalRead(6)) {
MAGNET("ON");
printf("Magnets!\n");
//if then it turns on light off the LED and return Magnets!
}
}
else if(0 ==digitalRead(6)) {
delay(20);
//If it is off then wait 20 miliseconds.
if(1 == digitalRead(0)) {
while(!digitalRead(0)) {
MAGNET("OFF");
}
//if there is suddenly a magnet read then while the read becomes the opposite, or
//zero, then return back that the magnet is off and keep the off LED on

}
}
}
}
